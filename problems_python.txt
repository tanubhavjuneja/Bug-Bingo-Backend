# Problem 6: Find the Largest Number

    largest = lst[0]

# Problem 11: Count Vowels
def count_vowels(s):
    vowels = "aeiou"
    count = 0
    for char in s:
        if char not in vowels:
            count += 1
    return count

print(count_vowels("hello"))

Expected Output:
2
